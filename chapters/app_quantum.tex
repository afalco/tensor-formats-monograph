\chapter{Operators on Hilbert Spaces: Mathematical Background}
\label{app:quantum}

This appendix collects the operator-theoretic foundations needed for
Part~V (Chapters~12 and~13). The presentation is self-contained and
mathematically rigorous; no prior knowledge of quantum mechanics is
assumed. All vector spaces are over $\CC$ in this appendix unless
stated otherwise. Standard references are~\cite{ReedSimon1972}
and~\cite{BratteliRobinson1987}; for the trace-class and
Hilbert-Schmidt theory from the perspective of tensor norms
see~\cite{Ryan2002}.

%------------------------------------------------------------
\section{Complex Hilbert spaces}
\label{app:quantum:hilbert}
%------------------------------------------------------------

\subsection*{Inner products and Hilbert spaces}

A \emph{complex Hilbert space} is a complex vector space $H$ equipped
with an \emph{inner product} $\langle \cdot, \cdot \rangle: H \times H
\to \CC$, which is:
\begin{enumerate}
\item[(IP1)] \emph{Sesquilinear}: linear in the second argument and
    conjugate-linear in the first, i.e.,
    $\langle u, \alpha v + \beta w\rangle = \alpha\langle u,v\rangle
    + \beta\langle u,w\rangle$ and
    $\langle \alpha u, v\rangle = \overline{\alpha}\langle u,v\rangle$.
\item[(IP2)] \emph{Hermitian}: $\langle u,v\rangle = \overline{\langle v,u\rangle}$.
\item[(IP3)] \emph{Positive definite}: $\langle u,u\rangle > 0$ for
    all $u \neq 0$.
\end{enumerate}
The induced norm $\|u\| \coloneqq \sqrt{\langle u,u\rangle}$ makes $H$
a normed space. When $H$ is complete with respect to this norm, it is
a \emph{Hilbert space}.

\begin{remark}[Convention on sesquilinearity]
\label{rem:app:sesqui}
The convention adopted here --- linearity in the second argument ---
is standard in mathematics. Physics texts often use the opposite
convention. Since the monograph works exclusively in the mathematical
convention, all statements below are consistent with it.
\end{remark}

\begin{example}
\label{ex:app:l2}
The space $\ell^2(\NN) \coloneqq \{(a_n)_{n \in \NN} \subset \CC :
\sum_n |a_n|^2 < \infty\}$ with $\langle a, b\rangle = \sum_n
\overline{a_n} b_n$ is a Hilbert space. More generally, for any
measure space $(\Omega, \mu)$, the space $L^2(\Omega, \mu; \CC)$
of square-integrable complex-valued functions with
$\langle f,g\rangle = \int_\Omega \overline{f}\, g\, d\mu$
is a Hilbert space.
\end{example}

Throughout Part~V and this appendix, $H$ denotes a
\emph{separable} complex Hilbert space, i.e., one that admits a
countable (finite or infinite) orthonormal basis.

\subsection*{Orthonormal bases}

A family $\{e_k\}_{k \in K}$ in $H$ is \emph{orthonormal} if
$\langle e_j, e_k\rangle = \delta_{jk}$. It is a \emph{(Hilbert)
orthonormal basis} if, in addition, its closed linear span is all of
$H$. Every separable Hilbert space has an orthonormal basis, and any
two orthonormal bases of $H$ have the same cardinality (finite or
countably infinite). If $H$ is infinite-dimensional and $\{e_k\}_{k\geq 1}$
is an orthonormal basis, every $u \in H$ has a unique expansion
\begin{equation}
\label{eq:app:fourier}
    u = \sum_{k=1}^\infty \langle e_k, u\rangle\, e_k,
    \qquad
    \|u\|^2 = \sum_{k=1}^\infty |\langle e_k, u\rangle|^2.
\end{equation}

\subsection*{The Riesz isomorphism}

By the Riesz representation theorem, for every bounded linear
functional $\varphi \in H^*$ there exists a unique $v_\varphi \in H$
such that $\varphi(u) = \langle v_\varphi, u\rangle$ for all $u \in H$.
The map $J: H \to H^*$, $J(v) \coloneqq \langle v, \cdot\rangle$,
is a conjugate-linear isometric isomorphism, called the \emph{Riesz
isomorphism}.

\begin{remark}[Riesz isomorphism vs.\ duality mapping]
\label{rem:app:riesz_duality}
In the real Banach space setting of Chapter~\ref{chap:banach}, the
duality mapping $J: X \to 2^{X^*}$ is in general set-valued and
nonlinear. In a complex Hilbert space, $J$ is single-valued and
conjugate-linear (hence $\RR$-linear). This simplification is the
reason why projection and best-approximation problems are simpler in
Hilbert spaces: every closed subspace is complemented via orthogonal
projection, and the metric projection onto a closed convex set is
given by the (linear) orthogonal projection when the set is a subspace.
\end{remark}

\begin{remark}[Dirac notation]
\label{rem:app:dirac}
In the physics literature, the inner product $\langle u, v\rangle$ is
written $\langle u | v\rangle$, with $u$ called a \emph{bra} and $v$
a \emph{ket}. An element $v \in H$ is written $|v\rangle$, its Riesz
dual $\langle v, \cdot\rangle \in H^*$ is written $\langle v|$, and
the rank-one operator $u \mapsto \langle v, u\rangle w$ is written
$|w\rangle\langle v|$. This notation, introduced by Dirac~\cite{Dirac1930},
is compact for computation but obscures the mathematical structure.
This monograph uses exclusively standard mathematical notation:
$\langle v, u\rangle$ for the inner product, $v \otimes \varphi$ for
rank-one operators (where $\varphi = J(v) \in H^*$), and
$P_S$ for the orthogonal projection onto a subspace $S \subset H$.
\end{remark}

\section{Bounded operators and the adjoint}
\label{app:quantum:bounded}
%------------------------------------------------------------

\subsection*{The operator algebra $\cL(H)$}

The space $\cL(H)$ of bounded linear operators $A: H \to H$, equipped
with the operator norm $\|A\| \coloneqq \sup_{\|u\| \leq 1} \|Au\|$,
is a Banach algebra: $\|AB\| \leq \|A\|\|B\|$ and $\|I\| = 1$.
Moreover, $\cL(H)$ is a \emph{C*-algebra}: every $A \in \cL(H)$ has
an \emph{adjoint} $A^* \in \cL(H)$ characterised by
\begin{equation}
\label{eq:app:adjoint}
    \langle A^* u, v\rangle = \langle u, Av\rangle
    \quad \forall\, u, v \in H,
\end{equation}
satisfying $\|A^*\| = \|A\|$ and $\|A^*A\| = \|A\|^2$ (C*-identity).

\subsection*{Special classes of operators}

\begin{definition}
\label{def:app:operator_classes}
An operator $A \in \cL(H)$ is:
\begin{enumerate}
\item[\textup{(i)}] \emph{Self-adjoint} if $A = A^*$.
\item[\textup{(ii)}] \emph{Positive} (written $A \geq 0$ or
    $A \succeq 0$) if $\langle u, Au\rangle \geq 0$ for all $u \in H$.
\item[\textup{(iii)}] \emph{Strictly positive} (written $A > 0$ or
    $A \succ 0$) if $\langle u, Au\rangle > 0$ for all $u \neq 0$.
\item[\textup{(iv)}] \emph{Unitary} if $A^*A = AA^* = I$.
\item[\textup{(v)}] An \emph{orthogonal projection} if $A^2 = A = A^*$.
\end{enumerate}
\end{definition}

\begin{proposition}
\label{prop:app:positive_equiv}
For $A \in \cL(H)$ self-adjoint, the following are equivalent:
\begin{enumerate}
\item[\textup{(a)}] $A \geq 0$.
\item[\textup{(b)}] $A = B^*B$ for some $B \in \cL(H)$.
\item[\textup{(c)}] The spectrum $\sigma(A) \subset [0, \infty)$.
\end{enumerate}
\end{proposition}

Every positive operator $A \geq 0$ has a unique positive square root
$A^{1/2} \geq 0$ with $(A^{1/2})^2 = A$; its \emph{absolute value}
is $|A| \coloneqq (A^*A)^{1/2}$.

\begin{proposition}[Orthogonal projections and subspaces]
\label{prop:app:proj_subspace}
There is a bijection between:
\begin{enumerate}
\item[\textup{(a)}] Closed subspaces $S \subset H$, and
\item[\textup{(b)}] Orthogonal projections $P \in \cL(H)$,
\end{enumerate}
given by $S = \mathrm{Im}(P)$. The orthogonal complement of $S$ is
$S^\perp = \ker(P)$, and $H = S \oplus S^\perp$ (orthogonal direct sum).
The operator $P_S \coloneqq P$ is called the \emph{orthogonal
projection onto $S$}.
\end{proposition}

\begin{remark}[Hilbert vs.\ Banach: complementability]
Every closed subspace of a Hilbert space is complemented
(Proposition~\ref{prop:app:proj_subspace}); its orthogonal complement
provides the split. As noted in Chapter~\ref{chap:banach}, this fails
in general Banach spaces and is the source of the genuine difficulty
in the immersion theorems of Chapters~\ref{chap:tucker_manifold}
and~\ref{chap:TB_manifold}.
\end{remark}

\subsection*{Rank-one operators}

For $u, v \in H$, the \emph{rank-one operator} $u \otimes \overline{v}$
is defined by
\begin{equation}
\label{eq:app:rank_one}
    (u \otimes \overline{v})(w)
    \coloneqq \langle v, w\rangle\, u,
    \quad w \in H.
\end{equation}
Its adjoint is $(u \otimes \overline{v})^* = v \otimes \overline{u}$,
and its norm is $\|u \otimes \overline{v}\| = \|u\|\|v\|$.
The orthogonal projection onto $\spann\{u\}$ (for $u \neq 0$) is
\begin{equation}
\label{eq:app:rank_one_proj}
    P_{\spann\{u\}}
    = \frac{u \otimes \overline{u}}{\|u\|^2}.
\end{equation}

\section{Compact operators and singular values}
\label{app:quantum:compact}
%------------------------------------------------------------

\subsection*{Compact operators}

An operator $A \in \cL(H)$ is \emph{compact} if the image of the
closed unit ball under $A$ is relatively compact (has compact closure).
The set $\cK(H) \subset \cL(H)$ of compact operators is a closed
two-sided ideal. Every finite-rank operator is compact; the closure
of finite-rank operators in $\cL(H)$ is $\cK(H)$ (when $H$ is
separable).

\begin{theorem}[Spectral theorem for compact self-adjoint operators]
\label{thm:app:spectral}
Let $A \in \cK(H)$ be self-adjoint. Then there exists an orthonormal
sequence $\{u_k\}_{k \geq 1}$ in $H$ (finite or infinite) and a
real sequence $(\lambda_k)$ with $\lambda_k \to 0$, such that
\begin{equation}
\label{eq:app:spectral}
    A = \sum_{k=1}^\infty \lambda_k\, u_k \otimes \overline{u_k},
\end{equation}
where the series converges in operator norm. The $\lambda_k$ are
the non-zero eigenvalues of $A$ (with multiplicity) and the $u_k$ the
corresponding eigenvectors. The decomposition~\eqref{eq:app:spectral}
is called the \emph{spectral decomposition} of $A$.
\end{theorem}

\subsection*{Singular values and singular value decomposition}

For any $A \in \cK(H)$, the operator $A^*A \in \cK(H)$ is compact
and positive, so its spectral theorem gives eigenvalues
$s_1(A)^2 \geq s_2(A)^2 \geq \cdots \geq 0$.

\begin{definition}
\label{def:app:singular_values}
The \emph{singular values} of $A \in \cK(H)$ are
$s_k(A) \coloneqq \lambda_k(|A|)^{1/2}$, where $|A| = (A^*A)^{1/2}$
and $\lambda_k(|A|)$ are the eigenvalues of $|A|$ in decreasing order.
\end{definition}

\begin{theorem}[Singular value decomposition]
\label{thm:app:svd}
For $A \in \cK(H)$ with singular values $s_1(A) \geq s_2(A) \geq
\cdots > 0$ there exist orthonormal sequences $\{u_k\}$ and $\{v_k\}$
in $H$ such that
\begin{equation}
\label{eq:app:svd}
    A = \sum_{k=1}^\infty s_k(A)\, v_k \otimes \overline{u_k},
\end{equation}
with convergence in operator norm. The $u_k$ are eigenvectors of $A^*A$
and the $v_k$ of $AA^*$.
\end{theorem}

\begin{remark}[Connection to Tucker rank]
\label{rem:app:svd_tucker}
For $d = 2$ and Hilbert spaces $H_1, H_2$, the SVD~\eqref{eq:app:svd}
is the Schmidt decomposition of a tensor in $H_1 \otimes_H H_2$
(Proposition~\ref{prop:app:hs_rank} in Appendix~\ref{app:functional}).
The number of non-zero singular values is the Tucker rank $r_1 = r_2 = r$.
An operator has finite rank $r$ if and only if it has exactly $r$
non-zero singular values.
\end{remark}

\section{Hilbert--Schmidt operators}
\label{app:quantum:hs}
%------------------------------------------------------------

\subsection*{Definition and basic properties}

\begin{definition}
\label{def:app:hs}
An operator $A \in \cL(H)$ is \emph{Hilbert--Schmidt} if
\begin{equation}
\label{eq:app:hs_norm}
    \|A\|_2^2 \coloneqq \sum_{k=1}^\infty \|A e_k\|^2 < \infty
\end{equation}
for some (equivalently, every) orthonormal basis $\{e_k\}$ of $H$.
The space of Hilbert--Schmidt operators is denoted $\cT{2}(H)$.
\end{definition}

The quantity $\|A\|_2$ is independent of the choice of orthonormal
basis and satisfies $\|A\|_2^2 = \Tr(A^*A) = \sum_k s_k(A)^2$.

\begin{proposition}
\label{prop:app:hs_properties}
The space $\cT{2}(H)$ of Hilbert--Schmidt operators satisfies:
\begin{enumerate}
\item[\textup{(i)}] $\cT{2}(H)$ is a Hilbert space under the
    \emph{Hilbert--Schmidt inner product}
    \begin{equation}
    \label{eq:app:hs_inner}
        \langle A, B\rangle_{\mathrm{HS}}
        \coloneqq \sum_{k=1}^\infty \langle A e_k, B e_k\rangle
        = \Tr(A^*B).
    \end{equation}
\item[\textup{(ii)}] $\cT{2}(H) \subset \cK(H)$ and
    $\|A\| \leq \|A\|_2$ for all $A \in \cT{2}(H)$.
\item[\textup{(iii)}] $\cT{2}(H)$ is a two-sided ideal in $\cL(H)$:
    if $A \in \cT{2}(H)$ and $B \in \cL(H)$, then
    $AB, BA \in \cT{2}(H)$ with $\|BA\|_2 \leq \|B\|\|A\|_2$.
\item[\textup{(iv)}] $A \in \cT{2}(H)$ if and only if
    $A^* \in \cT{2}(H)$, with $\|A^*\|_2 = \|A\|_2$.
\end{enumerate}
\end{proposition}

\subsection*{Tensor product identification}

The following identification connects Hilbert--Schmidt operators with
the Hilbert tensor product of Appendix~\ref{app:functional}
(Theorem~\ref{thm:app:hilbert_tensor_norms}).

\begin{theorem}
\label{thm:app:hs_tensor}
There is a canonical isometric isomorphism of Hilbert spaces
\begin{equation}
\label{eq:app:hs_tensor_iso}
    \cT{2}(H) \cong H \otimes_H \overline{H},
\end{equation}
where $\overline{H}$ denotes the \emph{conjugate Hilbert space}
\textup{(}same underlying set, same addition, scalar multiplication
by $\overline{\lambda}$, inner product
$\langle u, v\rangle_{\overline{H}} = \overline{\langle u,v\rangle_H}$\textup{)}.
Explicitly, the elementary tensor $u \otimes \overline{v}$
\textup{(}for $u \in H$, $\overline{v} \in \overline{H}$\textup{)}
corresponds to the rank-one operator $w \mapsto \langle v, w\rangle u$.
\end{theorem}

\begin{remark}[Hilbert--Schmidt operators as Tucker rank-$(r,r)$ tensors]
\label{rem:app:hs_tucker}
Under the identification~\eqref{eq:app:hs_tensor_iso}, a
Hilbert--Schmidt operator $A \in \cT{2}(H)$ of rank $r$ corresponds
to a tensor in $H \otimes_H \overline{H}$ of Tucker rank $(r,r)$
(Proposition~\ref{prop:app:hs_rank}).
Its singular value decomposition~\eqref{eq:app:svd} is precisely
the Tucker decomposition of this tensor:
\begin{equation*}
    A = \sum_{k=1}^r s_k(A)\, v_k \otimes \overline{u_k}
    \in H \otimes_H \overline{H},
\end{equation*}
with $\{u_k\}$, $\{v_k\}$ orthonormal bases of the minimal subspaces
$\Umin{1}{A} = \overline{\mathrm{Im}(A^*)}$ and
$\Umin{2}{A} = \overline{\mathrm{Im}(A)}$ respectively.
\end{remark}

\section{Trace-class operators}
\label{app:quantum:trace}
%------------------------------------------------------------

\subsection*{Definition and trace norm}

\begin{definition}
\label{def:app:trace_class}
An operator $A \in \cL(H)$ is \emph{trace class} if
\begin{equation}
\label{eq:app:trace_norm}
    \|A\|_1 \coloneqq \Tr\,|A|
    = \sum_{k=1}^\infty s_k(A) < \infty.
\end{equation}
The space of trace-class operators is denoted $\cT{1}(H)$.
\end{definition}

\begin{proposition}[Trace]
\label{prop:app:trace}
For $A \in \cT{1}(H)$ and any orthonormal basis $\{e_k\}$ of $H$,
the series
\begin{equation}
\label{eq:app:trace}
    \Tr(A) \coloneqq \sum_{k=1}^\infty \langle e_k, A e_k\rangle
\end{equation}
converges absolutely and is independent of the choice of basis.
\end{proposition}

The trace is the unique linear functional on $\cT{1}(H)$ satisfying
$\Tr(AB) = \Tr(BA)$ for all $A \in \cT{1}(H)$,
$B \in \cL(H)$, and $\Tr(P) = \dim \mathrm{Im}(P)$ for every
finite-rank orthogonal projection $P$.

\subsection*{Properties of $\cT{1}(H)$}

\begin{proposition}
\label{prop:app:trace_class_properties}
The space $\cT{1}(H)$ of trace-class operators satisfies:
\begin{enumerate}
\item[\textup{(i)}] $\cT{1}(H) \subset \cT{2}(H) \subset \cK(H)$, with
    $\|A\| \leq \|A\|_2 \leq \|A\|_1$.
\item[\textup{(ii)}] $(\cT{1}(H), \|\cdot\|_1)$ is a Banach space
    \textup{(}not a Hilbert space\textup{)}.
\item[\textup{(iii)}] $\cT{1}(H)$ is a two-sided ideal in $\cL(H)$:
    $\|BA\|_1 \leq \|B\|\|A\|_1$ for $A \in \cT{1}(H)$, $B \in \cL(H)$.
\item[\textup{(iv)}] \emph{Duality:} The trace pairing
    $\langle A, X\rangle \coloneqq \Tr(A^*X)$
    identifies $(\cT{1}(H))^* \cong \cL(H)$ isometrically.
\item[\textup{(v)}] The spectral decomposition of a self-adjoint
    $A \in \cT{1}(H)$ is~\eqref{eq:app:spectral} with
    $\sum_k |\lambda_k| < \infty$, and
    $\|A\|_1 = \sum_k |\lambda_k|$.
\end{enumerate}
\end{proposition}

\begin{remark}[Trace-class as Banach tensor norm]
\label{rem:app:trace_projective}
The trace norm $\|\cdot\|_1$ on $\cT{1}(H)$ coincides with the
\emph{projective tensor norm} $\|\cdot\|_\pi$ on
$H \otimes_\pi \overline{H}$ (under the identification
$\cT{1}(H) \cong H \otimes_\pi \overline{H}$). This is the dual to
the injective norm: $(\cT{1}(H))^* \cong \cK(H)$ under the Hilbert--Schmidt
metric and $(\cT{1}(H))^* \cong \cL(H)$ under the trace pairing.
In the language of Chapter~\ref{chap:algtensor}:
\begin{equation*}
    \|\cdot\|_\varepsilon \leq \|\cdot\|_H
    \leq \|\cdot\|_\pi = \|\cdot\|_1
    \quad \text{on } H \otimes \overline{H},
\end{equation*}
where $\|\cdot\|_H$ is the Hilbert tensor norm and $\|\cdot\|_\varepsilon$
is the injective norm. The trace norm is the natural norm for
quantum states because it controls distinguishability: for two states
$\rho$ and $\sigma$, $\frac{1}{2}\|\rho - \sigma\|_1$ is the
\emph{total variation distance}, the maximum probability of distinguishing
$\rho$ from $\sigma$ by any quantum measurement.
\end{remark}

\subsection*{Partial trace}

Let $H = H_1 \otimes_H H_2$ be a Hilbert tensor product. For
$A \in \cT{1}(H)$, the \emph{partial trace} over $H_2$ is the unique
operator $\TrS{2}(A) \in \cT{1}(H_1)$ satisfying
\begin{equation}
\label{eq:app:partial_trace}
    \TrS{H_1}\bigl(B\, \TrS{2}(A)\bigr)
    = \Tr_{H}\bigl((B \otimes I_2)\, A\bigr)
    \quad \forall\, B \in \cL(H_1).
\end{equation}
For an orthonormal basis $\{f_j\}$ of $H_2$:
\begin{equation*}
    \TrS{2}(A) = \sum_j (I_1 \otimes f_j^*)
    \, A\, (I_1 \otimes f_j) \in \cT{1}(H_1).
\end{equation*}

\begin{proposition}[Partial trace and minimal subspaces]
\label{prop:app:partial_trace_min}
Let $v \in H_1 \otimes_H H_2$ with $\|v\|=1$ and let
$A_v = v \otimes \overline{v} \in \cT{1}(H)$ be the associated
rank-one operator. Then:
\begin{enumerate}
\item[\textup{(a)}] $\TrS{2}(A_v) \in \cT{1}(H_1)$ is
    a positive operator with trace $1$ and
    \begin{equation*}
        \supp\bigl(\TrS{2}(A_v)\bigr)
        = \Umin{1}{v},
    \end{equation*}
    where $\Umin{1}{v}$ is the minimal subspace of $v$ in the
    direction $\{1\}$ \textup{(}Definition~\ref{def:minimal_subspace}\textup{)}.
\item[\textup{(b)}] $\mathrm{rank}\bigl(\TrS{2}(A_v)\bigr)
    = r_1(v)$, the Tucker $1$-rank of $v$.
\item[\textup{(c)}] The non-zero eigenvalues of $\TrS{2}(A_v)$
    are $s_k(v)^2 / \|v\|^2$, where $s_k(v)$ are the singular values
    of $v$ viewed as an operator $H_1 \to H_2$.
\end{enumerate}
\end{proposition}

\begin{proof}
Write $v = \sum_{k=1}^r s_k\, u_k \otimes w_k$ (Schmidt decomposition,
Theorem~\ref{thm:app:svd} applied to $v$ as an operator $H_1 \to H_2$).
Then
\begin{equation*}
    A_v = v \otimes \overline{v}
    = \sum_{j,k} s_j s_k\,
    (u_j \otimes \overline{u_k}) \otimes (w_j \otimes \overline{w_k}).
\end{equation*}
Applying the partial trace:
\begin{align*}
    \TrS{2}(A_v)
    &= \sum_{j,k} s_j s_k\, (u_j \otimes \overline{u_k})
       \sum_l \langle w_l, w_j\rangle\langle w_k, w_l\rangle \\
    &= \sum_{j,k} s_j s_k\, (u_j \otimes \overline{u_k})\, \delta_{jk}
     = \sum_k s_k^2\, u_k \otimes \overline{u_k}.
\end{align*}
This has support $\spann\{u_k : s_k > 0\} = \Umin{1}{v}$ and rank $r_1(v)$.
\end{proof}

\section{Density operators and quantum states}
\label{app:quantum:states}
%------------------------------------------------------------

\subsection*{The space of states}

\begin{definition}
\label{def:app:density}
Let $H$ be a separable complex Hilbert space. An operator
$\rho \in \cTsa{1}(H) \coloneqq \{A \in \cT{1}(H) : A = A^*\}$
is a \emph{density operator} (or \emph{quantum state}) if
\begin{enumerate}
\item[(i)] $\rho \geq 0$ \quad (positivity),
\item[(ii)] $\Tr(\rho) = 1$ \quad (normalisation).
\end{enumerate}
The set of all density operators is
\begin{equation}
\label{eq:app:state_space}
    \Dstate{H}
    \coloneqq \bigl\{
    \rho \in \cTsa{1}(H) : \rho \geq 0,\;
    \Tr(\rho) = 1
    \bigr\}.
\end{equation}
\end{definition}

$\Dstate{H}$ is a convex subset of the affine hyperplane
$\{\Tr = 1\} \subset \cTsa{1}(H)$.

\subsection*{Pure and mixed states}

\begin{definition}
\label{def:app:pure_mixed}
A density operator $\rho \in \Dstate{H}$ is:
\begin{enumerate}
\item[\textup{(i)}] \emph{Pure} if $\mathrm{rank}(\rho) = 1$, equivalently
    $\rho^2 = \rho$ (it is an orthogonal projection of rank one).
\item[\textup{(ii)}] \emph{Mixed} if $\mathrm{rank}(\rho) \geq 2$.
\end{enumerate}
\end{definition}

Every pure state has the form $\rho = v \otimes \overline{v}$ for some
unit vector $v \in H$. Two unit vectors $v$ and $e^{i\theta}v$ ($\theta
\in \RR$) give the same pure state. Thus the set of pure states is in
bijection with the \emph{projective Hilbert space}
$\mathbb{P}(H) \coloneqq (H \setminus \{0\})/\CC^*$.

Every mixed state $\rho$ has a spectral decomposition
\begin{equation}
\label{eq:app:spectral_state}
    \rho = \sum_{k=1}^r \lambda_k\, v_k \otimes \overline{v_k},
\end{equation}
where $r = \mathrm{rank}(\rho)$, $\{v_k\}_{k=1}^r$ is an orthonormal
system, and $\lambda_k > 0$ with $\sum_k \lambda_k = 1$.

\begin{remark}[The fundamental connection with minimal subspaces]
\label{rem:app:states_minimal}
Let $v \in H$ be non-zero. The rank-one operator
\begin{equation}
\label{eq:app:pure_state}
    \rho_v \coloneqq \frac{v \otimes \overline{v}}{\|v\|^2}
    \in \Dstate{H}
\end{equation}
is the pure state associated to $v$.
Its \emph{support} $\supp(\rho_v) = \mathrm{Im}(\rho_v)
= \spann\{v\}$ is exactly the minimal subspace
$\Umin{\{1,\ldots,d\}}{v}$ of $v$ in the full index set $D$
(Definition~\ref{def:minimal_subspace}).

In a multipartite system $H = \bigotimes_{j=1}^d H_j$, the
partial trace $\rho_\alpha \coloneqq \Tr_{\alpha^c}(\rho_v)$
satisfies, by Proposition~\ref{prop:app:partial_trace_min}:
\begin{equation}
\label{eq:app:supp_minimal}
    \supp(\rho_\alpha) = \Umin{\alpha}{v},
    \qquad
    \mathrm{rank}(\rho_\alpha) = r_\alpha(v).
\end{equation}
This is the cornerstone of Chapter~12: the
Tucker rank vector $(r_\alpha(v))_{\alpha \in D}$ encodes the
\emph{entanglement structure} of the pure state $\rho_v$.
\end{remark}

\subsection*{The rank stratification}

\begin{definition}
\label{def:app:rank_strata}
For $r \in \NN$, the \emph{rank-$r$ stratum} of $\Dstate{H}$ is
\begin{equation}
\label{eq:app:rank_stratum}
    \Drank{r}{H}
    \coloneqq \bigl\{ \rho \in \Dstate{H} :
    \mathrm{rank}(\rho) = r \bigr\}.
\end{equation}
\end{definition}

The full state space decomposes as a disjoint union:
\begin{equation}
\label{eq:app:state_stratification}
    \Dstate{H} = \bigsqcup_{r=1}^\infty \Drank{r}{H}
    \cup \mathcal{D}_\infty(H),
\end{equation}
where $\mathcal{D}_\infty(H)$ denotes the infinite-rank states.
This mirrors the disjoint decomposition $\VD = \bigsqcup_\fr
\Tucker{\fr}{\VD}$ of Theorem~\ref{thm:tucker_decomposition_full}.

\begin{proposition}[Closure relations]
\label{prop:app:state_closure}
In the trace-norm topology:
\begin{enumerate}
\item[\textup{(i)}] $\overline{\Drank{r}{H}}^{\|\cdot\|_1}
    = \bigcup_{k=1}^r \mathcal{D}_k(H)$
    \textup{(}rank is lower semicontinuous\textup{)}.
\item[\textup{(ii)}] $\mathcal{D}_k(H) \subset
    \overline{\Drank{r}{H}}^{\|\cdot\|_1}$ for all $1 \leq k \leq r$.
\end{enumerate}
\end{proposition}

These closure relations are the quantum analogues of the weak lower
semicontinuity of Tucker rank established in
Theorem~\ref{thm:TBT_rank_lsc}.

\section{Positive core and the support decomposition}
\label{app:quantum:core}
%------------------------------------------------------------

This section develops the local parametrisation of $\Drank{r}{H}$
that will be used in Chapter~12.

\subsection*{The positive core manifold}

For an $r$-dimensional subspace $S \in \Grass{r}{H}$, define
\begin{equation}
\label{eq:app:pos_core}
    \Dpos{S}
    \coloneqq \bigl\{
    \sigma \in \cTsa{1}(S) : \sigma > 0,\;
    \TrS{S}(\sigma) = 1
    \bigr\}.
\end{equation}

\begin{proposition}
\label{prop:app:pos_core_manifold}
$\Dpos{S}$ is an open convex subset of the affine hyperplane
$\{\TrS{S} = 1\} \subset \cTsa{1}(S) \cong \RR^{r^2 - 1}$,
hence a smooth finite-dimensional manifold of real dimension $r^2 - 1$.
\end{proposition}

\begin{remark}[Core as Tucker format over $\RR$]
\label{rem:app:core_tucker}
In the notation of Chapter~\ref{chap:algtensor}, a density operator
$\sigma \in \Dpos{S}$ is (after choosing a basis of $S$) an
element of $M_r(\CC)^{sa}_+ \cap \{\Tr = 1\}$, i.e., a positive
definite Hermitian matrix of trace 1. This is the finite-dimensional
fibre in the Tucker bundle structure of $\Drank{r}{H}$, playing the
role of the core tensor $C^{(D)} \in \RR^{\times r_\alpha}_*$ from
Chapter~\ref{chap:tucker_manifold}.
\end{remark}

\subsection*{The support map and local coordinates}

The \emph{support map}
\begin{equation}
\label{eq:app:support_map}
    \pi: \Drank{r}{H} \longrightarrow \Grass{r}{H},
    \qquad
    \pi(\rho) \coloneqq \supp(\rho) = \mathrm{Im}(\rho),
\end{equation}
is the quantum analogue of the rank map $\varrho_\fr$ of
Chapter~\ref{chap:tucker_manifold}.

\begin{proposition}[Local coordinates for $\Drank{r}{H}$]
\label{prop:app:local_coords}
Fix $\rho_0 \in \Drank{r}{H}$ and set $S_0 = \supp(\rho_0)$.
Let $V_0: \CC^r \to H$ be an isometry with $\mathrm{Im}(V_0) = S_0$.
For $A \in \cL(\CC^r, S_0^\perp)$ sufficiently small, define the
isometric embedding
\begin{equation}
\label{eq:app:va}
    V_A \coloneqq (V_0 + AV_0)(I + V_0^*A^*AV_0)^{-1/2}
    \in \cL(\CC^r, H), \quad V_A^* V_A = I.
\end{equation}
Then the map
\begin{equation}
\label{eq:app:chart_states}
    \Phi_{S_0}: (A, \sigma)
    \longmapsto V_A \sigma V_A^*
    \in \Drank{r}{H}
\end{equation}
is a $\mathcal{C}^\infty$-diffeomorphism from a neighbourhood of
$(0, V_0^*\rho_0 V_0)$ in
$\cL(\CC^r, S_0^\perp) \times \mathcal{D}_+(S_0^\perp\text{-free})$
onto a neighbourhood of $\rho_0$ in $\Drank{r}{H}$.
\end{proposition}

This local chart is the quantum specialisation of the Tucker manifold
chart map (Lemma~\ref{lem:element}): the Grassmann variable $A$ encodes
the motion of the support subspace, and the core variable $\sigma$
encodes the positive degrees of freedom within that subspace.

\section{Historical notes}
\label{app:quantum:history}
%------------------------------------------------------------

\paragraph{Mathematical foundations of quantum mechanics.}
The description of quantum states by operators on Hilbert spaces was
introduced by von Neumann~\cite{vonNeumann1932}, whose 1932 monograph
established the functional-analytic framework of quantum mechanics.
The use of density operators (rather than pure state vectors) to
describe mixed states goes back to von Neumann's 1927 papers on
entropy and measurement.

\paragraph{Trace-class operators.}
The systematic study of trace-class operators as a Banach space was
developed by Schatten~\cite{Schatten1950}, who introduced the ideals
$\cT{p}(H)$ (now called \emph{Schatten $p$-classes}) with norm
$\|A\|_p = (\sum_k s_k(A)^p)^{1/p}$. The cases $p=1$ (trace class)
and $p=2$ (Hilbert--Schmidt) are the most important. The duality
$(\cT{1}(H))^* \cong \cL(H)$ is the non-commutative analogue of the
duality $(\ell^1)^* \cong \ell^\infty$.

\paragraph{Tensor norms and the trace.}
The identification $\cT{1}(H) \cong H \otimes_\pi \overline{H}$
(Remark~\ref{rem:app:trace_projective}) was established by
Grothendieck~\cite{Grothendieck1953} as part of his study of tensor
norms. This places the trace norm firmly within the framework of
Chapter~\ref{chap:algtensor}: it is the projective tensor norm on
$H \otimes \overline{H}$, the largest reasonable crossnorm.
The Hilbert--Schmidt norm is the Hilbert tensor norm, which lies
strictly between the projective and injective norms when $H$ is
infinite-dimensional (Theorem~\ref{thm:app:hilbert_tensor_norms}).

\paragraph{Minimal subspaces and partial traces.}
The identification of the support of the partial trace with the
minimal subspace (Proposition~\ref{prop:app:partial_trace_min} and
Remark~\ref{rem:app:states_minimal}) appears implicitly in the quantum
information literature as the \emph{Schmidt rank}, introduced for
bipartite pure states by Schmidt (1907) long before the tensor
formalism. The generalisation to multipartite systems and the
systematic treatment as minimal subspaces is the contribution of
Falcó and Hackbusch~\cite{FH2012}, which is the foundation of
Chapters~\ref{chap:minsubspaces} and~12 of
this monograph.

\paragraph{The Grassmann-Banach structure of $\Drank{r}{H}$.}
The local chart structure of Proposition~\ref{prop:app:local_coords}
— decomposing a neighbourhood of $\rho$ in $\Drank{r}{H}$ into a
Grassmannian direction (the support) and a finite-dimensional core
(the positive operator on the support) — is developed in full detail
in the companion paper~\cite{AcevedoFalco2026}, whose results form
the mathematical content of Chapters~12
and~13.
